    /**
     * Toggles the theme between dark and light.
     */
    setTheme(newTheme) {
        localStorage.katanaTheme = newTheme;
        Object.keys(this.themes[newTheme]).forEach(key => {
            document.documentElement.style.setProperty(`--${key}`, this.themes[newTheme][key]);
        });
    }
